c++ algorithm tips!

* Vector
 #include <vector> //헤더파일
 Vector <type> vec; //선언
 vec.push_back(type); //배열처럼 순차적으로 추가
 Vec.begin(), vec.end() //벡터 시작, 끝
 Vec.size() //현재 벡터안에 원소의 수 capacity 와는 다른것!
 
* pair
 #include <utility> //헤더파일
 Pair<type, type> p //선언
 P = make_pair(x, y) //이런식으로 할당 가능

*sort
 #include <algorithm> //헤더파일
 sort(arr,arr+N) //배열의 경우
 sort(vec.begin(), vec.end()) //벡터의 경우
 sort의 세번째에 비교 함수를 내가 정의해 쓸 수 있다
 -> bool cmp(pair<int, int> a, pair<int, int> b){
	return a.first < b.first; //오름 차순일 때
    }
